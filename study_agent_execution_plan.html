async function updateUserProgress(userId, assignmentId, exercise, evaluation) { let progress = await UserProgress.findOne({ userId, assignmentId }); if (!progress) { progress = new UserProgress({ userId, assignmentId, topicMastery: {}, weakTopics: [], strongTopics: [] }); } // Update topic-specific mastery const topic = exercise.topic; if (!progress.topicMastery[topic]) { progress.topicMastery[topic] = { correct: 0, total: 0, averageDifficulty: 0, lastPracticed: new Date() }; } const topicData = progress.topicMastery[topic]; topicData.total += 1; if (evaluation.isCorrect) topicData.correct += 1; topicData.averageDifficulty = (topicData.averageDifficulty * (topicData.total - 1) + exercise.difficulty) / topicData.total; topicData.lastPracticed = new Date(); // Calculate mastery percentage const masteryPercentage = (topicData.correct / topicData.total) * 100; // Categorize as weak or strong if (masteryPercentage < 60 && topicData.total >= 3) { if (!progress.weakTopics.includes(topic)) { progress.weakTopics.push(topic); } progress.strongTopics = progress.strongTopics.filter(t => t !== topic); } else if (masteryPercentage >= 80 && topicData.total >= 5) { if (!progress.strongTopics.includes(topic)) { progress.strongTopics.push(topic); } progress.weakTopics = progress.weakTopics.filter(t => t !== topic); } // Calculate overall readiness const allTopics = Object.keys(progress.topicMastery); const totalMastery = allTopics.reduce((sum, t) => { const data = progress.topicMastery[t]; return sum + (data.correct / data.total); }, 0); progress.overallReadiness = Math.round((totalMastery / allTopics.length) * 100); progress.updatedAt = new Date(); await progress.save(); return progress; } function calculateNextDifficulty(topicData, lastExerciseDifficulty, wasCorrect) { // Adaptive difficulty: increase if correct, decrease if wrong let newDifficulty = lastExerciseDifficulty; if (wasCorrect) { newDifficulty = Math.min(5, lastExerciseDifficulty + 0.5); } else { newDifficulty = Math.max(1, lastExerciseDifficulty - 0.5); } // But don't let it drift too far from average mastery const masteryPercentage = (topicData.correct / topicData.total) * 100; const targetDifficulty = 1 + (masteryPercentage / 100) * 4; // Blend current difficulty with target return Math.round((newDifficulty * 0.7 + targetDifficulty * 0.3) * 2) / 2; }
